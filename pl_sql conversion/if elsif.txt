CREATE OR REPLACE PROCEDURE SELECT1(DESIGNATION IN VARCHAR2,BONUS_AMT OUT NUMBER)
IS
BEGIN
IF DESIGNATION='MANAGER' THEN 
BONUS_AMT:=4000;
ELSIF DESIGNATION='TEAM LEAD' THEN 
BONUS_AMT:=3000;
ELSE
BONUS_AMT:=2000;
END IF;
END;
SELECT * FROM EMPLOYEES;
SET SERVEROUTPUT ON
DECLARE
BONUS_AMT NUMBER;
BEGIN
SELECT1('TEAM LEAD',BONUS_AMT);
DBMS_OUTPUT.PUT_LINE(BONUS_AMT);
END;
