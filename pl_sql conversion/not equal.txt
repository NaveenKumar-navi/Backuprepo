CREATE OR REPLACE PROCEDURE SELECT1(EMP_ID IN NUMBER DEFAULT NULL,EMP_NAME OUT VARCHAR2,EMP_DOB OUT DATE)
IS
FLAG NUMBER;
BEGIN
FLAG:=0;
IF EMP_ID IS NULL OR FLAG!=0 THEN
EMP_NAME:=NULL;
EMP_DOB:=NULL;
ELSE
SELECT DECODE(NAME,NULL,'EMPLOYEE NAME NOT LOADED',NAME),DOB INTO EMP_NAME,EMP_DOB FROM EMPLOYEES WHERE ID=EMP_ID;
END IF;
END;
SET SERVEROUTPUT ON
DECLARE
EMP_NAME VARCHAR2(30);
EMP_DOB DATE;
BEGIN
SELECT1(NULL,EMP_NAME,EMP_DOB);
DBMS_OUTPUT.PUT_LINE(EMP_NAME||'DATE OF BIRTH IS'||EMP_DOB);
END;
