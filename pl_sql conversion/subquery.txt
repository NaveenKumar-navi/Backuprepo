CREATE OR REPLACE PROCEDURE SELECT1(EMP_ID IN NUMBER,SALARY OUT NUMBER)
IS
BEGIN
SELECT SAL INTO SALARY FROM SALARY S 
WHERE ID=EMP_ID 
    AND EXISTS (SELECT 1 FROM EMPLOYEES 
                WHERE ID=S.ID
                 AND NAME IS NOT NULL);
END;
