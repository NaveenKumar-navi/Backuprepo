CREATE OR REPLACE PROCEDURE SELECT1(EMP_ID NUMBER,EXISTSING OUT NUMBER)
IS
BEGIN
SELECT 1 INTO EXISTSING FROM EMPLOYEES S 
WHERE ID=EMP_ID
    AND EXISTS (SELECT 1 FROM EMPLOYEES E
                WHERE E.ID=S.ID
                 AND NAME IS NOT NULL);
END;
DECLARE
EXISTSING NUMBER;
BEGIN
SELECT1(3,EXISTSING);
DBMS_OUTPUT.PUT_LINE(EXISTSING);
END;
