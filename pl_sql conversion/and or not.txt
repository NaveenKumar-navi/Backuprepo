CREATE OR REPLACE PROCEDURE SELECT1(EMP_NAME OUT VARCHAR2)
IS
BEGIN
SELECT NAME INTO EMP_NAME FROM EMPLOYEES WHERE ((ID>4) AND (DESIGNATION='EXECUTIVE')) OR ((NAME IS NOT NULL) AND (ID>4));
END;
SET SERVEROUTPUT ON
DECLARE
EMP_NAME VARCHAR2(20);
BEGIN
SELECT1(EMP_NAME);
DBMS_OUTPUT.PUT_LINE(EMP_NAME);
END;
