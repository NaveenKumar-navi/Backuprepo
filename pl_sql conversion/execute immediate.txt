--------------EXECUTE IMMEDIATE------------
CREATE OR REPLACE PROCEDURE SELECT1
IS
ID NUMBER:=1;
SQL_STMT VARCHAR2(200);
BEGIN
   EXECUTE IMMEDIATE 'CREATE TABLE EXEC_IMMED(ID NUMBER)';
    SQL_STMT:='INSERT INTO EXEC_IMMED VALUES(:ID)';
    EXECUTE IMMEDIATE SQL_STMT USING ID;
END;
BEGIN
SELECT1;
END;
SELECT * FROM EXEC_IMMED;
