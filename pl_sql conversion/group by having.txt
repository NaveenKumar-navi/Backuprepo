CREATE OR REPLACE PROCEDURE SELECT1(EMP_DESIG OUT VARCHAR2,SUM_SAL OUT NUMBER)
IS
AVERAGE NUMBER;
BEGIN
SELECT DESIGNATION,SUM(SAL) INTO EMP_DESIG,SUM_SAL FROM SALARY S JOIN EMPLOYEES E
        ON S.ID=E.ID
        GROUP BY DESIGNATION
        HAVING SUM(SAL)>50000;
END;

SET SERVEROUTPUT ON
DECLARE
EMP_DESIG VARCHAR2(30);
SUM_SAL NUMBER;
BEGIN
SELECT1(EMP_DESIG,SUM_SAL);
DBMS_OUTPUT.PUT_LINE(EMP_DESIG||'                '||SUM_SAL);
END;
