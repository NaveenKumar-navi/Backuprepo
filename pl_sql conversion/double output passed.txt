CREATE OR REPLACE PROCEDURE SELECT1(EMP_ID NUMBER,EMP_NAME OUT VARCHAR2,EMP_DOB OUT DATE)
IS
BEGIN
SELECT DECODE(NAME,NULL,'EMPLOYEE NAME NOT LOADED',NAME),DOB INTO EMP_NAME,EMP_DOB FROM EMPLOYEES WHERE ID=EMP_ID;
END;
SET SERVEROUTPUT ON
DECLARE
EMP_NAME VARCHAR2(30);
EMP_DOB DATE;
BEGIN
SELECT1(6,EMP_NAME,EMP_DOB);
DBMS_OUTPUT.PUT_LINE(EMP_NAME||'DATE OF BIRTH IS'||EMP_DOB);
END;
