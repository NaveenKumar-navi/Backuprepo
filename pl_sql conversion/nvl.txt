-----------NVL-----------
CREATE OR REPLACE PROCEDURE SELECT1(EMP_ID NUMBER,EMP_NAME OUT VARCHAR2)
IS
BEGIN
SELECT NVL(NAME,'EMPLOYEE NAME NOT LOADED') INTO EMP_NAME FROM EMPLOYEES WHERE ID=EMP_ID;
END;
SET SERVEROUTPUT ON
DECLARE
EMP_NAME VARCHAR2(30);
BEGIN
SELECT1(6,EMP_NAME);
DBMS_OUTPUT.PUT_LINE(EMP_NAME);
END;
SELECT * FROM EMPLOYEES;
